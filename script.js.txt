// NAVEGACIÓN PRINCIPAL DEL OVA
function switchTab(tabId) {
    document.querySelectorAll('.tab-content').forEach(el => el.classList.remove('active-content'));
    document.querySelectorAll('.tab-btn').forEach(el => el.classList.remove('active'));
    
    document.getElementById('tab-' + tabId).classList.add('active-content');
    event.currentTarget.classList.add('active');
}

// ACORDEÓN REPOSITORIO
function toggleAccordion(element) {
    const body = element.nextElementSibling;
    const isVisible = body.style.display === 'block';
    body.style.display = isVisible ? 'none' : 'block';
    element.querySelector('span').textContent = isVisible ? '▼' : '▲';
}

// FORO INTERACTIVO
function postToForum() {
    const name = document.getElementById('forum-name').value.trim();
    const reply = document.getElementById('forum-reply').value.trim();
    
    if(!name || !reply) { alert("Por favor complete todos los campos."); return; }
    
    const container = document.getElementById('forum-comments-container');
    const newComment = document.createElement('div');
    newComment.className = 'comment';
    newComment.innerHTML = `<div class="comment-author">${name}</div><p>${reply}</p>`;
    container.prepend(newComment);
    
    document.getElementById('forum-name').value = '';
    document.getElementById('forum-reply').value = '';
}

// JUEGO 1: ORDENAR NIVELES MARCO CDD
const correctSortOrder = ["A1", "A2", "B1", "B2", "C1", "C2"];
let userSortOrder = [];
function initSortGame() {
    const container = document.getElementById('sort-container');
    container.innerHTML = '';
    userSortOrder = [];
    document.getElementById('sort-feedback').textContent = '';
    
    const shuffled = [...correctSortOrder].sort(() => Math.random() - 0.5);
    shuffled.forEach(level => {
        const btn = document.createElement('button');
        btn.className = 'btn-choice';
        btn.textContent = level;
        btn.onclick = () => {
            btn.classList.add('disabled');
            btn.disabled = true;
            userSortOrder.push(level);
            
            if(userSortOrder.length === correctSortOrder.length) {
                if(JSON.stringify(userSortOrder) === JSON.stringify(correctSortOrder)) {
                    document.getElementById('sort-feedback').textContent = "🎉 ¡Perfecto! Dominas los niveles de competencia desde el acceso hasta el liderazgo.";
                    document.getElementById('sort-feedback').style.color = "green";
                } else {
                    document.getElementById('sort-feedback').textContent = "❌ Orden incorrecto. Inténtalo de nuevo presionando 'Mód 1' arriba.";
                    document.getElementById('sort-feedback').style.color = "red";
                }
            }
        };
        container.appendChild(btn);
    });
}

// JUEGO 2: RELACIONAR CONCEPTOS
const matchPairs = [
    {c: "Pertinencia Pedagógica", d: "Alineación directa con los propósitos del Currículo."},
    {c: "Repositorio Educativo", d: "Espacio digital estructurado que aloja recursos abiertos."},
    {c: "Curación de Contenidos", d: "Filtrar, seleccionar y adaptar lo mejor de internet."}
];
let selectedConcept = null;
let selectedDef = null;

function initMatchGame() {
    const conDiv = document.getElementById('match-concepts');
    const defDiv = document.getElementById('match-definitions');
    conDiv.innerHTML = ''; defDiv.innerHTML = '';
    document.getElementById('match-feedback').textContent = '';

    matchPairs.forEach((pair, idx) => {
        const cEl = document.createElement('div');
        cEl.className = 'match-item concept';
        cEl.textContent = pair.c;
        cEl.dataset.idx = idx;
        cEl.onclick = () => selectItem(cEl, 'concept');
        conDiv.appendChild(cEl);
    });

    [...matchPairs].sort(() => Math.random() - 0.5).forEach((pair) => {
        const dEl = document.createElement('div');
        dEl.className = 'match-item definition';
        dEl.textContent = pair.d;
        // Encontrar índice original
        const origIdx = matchPairs.findIndex(p => p.d === pair.d);
        dEl.dataset.idx = origIdx;
        dEl.onclick = () => selectItem(dEl, 'definition');
        defDiv.appendChild(dEl);
    });
}

function selectItem(el, type) {
    if(el.classList.contains('correct')) return;
    
    if(type === 'concept') {
        document.querySelectorAll('.match-item.concept').forEach(i => i.classList.remove('selected'));
        selectedConcept = el;
    } else {
        document.querySelectorAll('.match-item.definition').forEach(i => i.classList.remove('selected'));
        selectedDef = el;
    }
    el.classList.add('selected');

    if(selectedConcept && selectedDef) {
        if(selectedConcept.dataset.idx === selectedDef.dataset.idx) {
            selectedConcept.classList.add('correct');
            selectedDef.classList.add('correct');
            selectedConcept.classList.remove('selected');
            selectedDef.classList.remove('selected');
            selectedConcept = null; selectedDef = null;
            
            if(document.querySelectorAll('.match-item.correct').length === matchPairs.length * 2) {
                document.getElementById('match-feedback').textContent = "🎉 ¡Excelente! Relacionaste todos los criterios pedagógicos correctamente.";
            }
        } else {
            setTimeout(() => {
                selectedConcept.classList.remove('selected');
                selectedDef.classList.remove('selected');
                selectedConcept = null; selectedDef = null;
            }, 500);
        }
    }
}

function renderQuiz() {
    const container = document.getElementById('quiz-container');
    container.innerHTML = '';
    quizData.forEach((item, qIdx) => {
        const itemDiv = document.createElement('div');
        itemDiv.className = 'quiz-item';
        itemDiv.innerHTML = `<p><strong>${item.q}</strong></p><div class="quiz-options"></div>`;
        const optionsDiv = itemDiv.querySelector('.quiz-options');
        
        item.a.forEach((opt, oIdx) => {
            optionsDiv.innerHTML += `
                <label>
                    <input type="radio" name="q${qIdx}" value="${oIdx}"> ${opt}
                </label>
            `;
        });
        container.appendChild(itemDiv);
    });
}

function gradeQuiz() {
    let score = 0;
    quizData.forEach((item, qIdx) => {
        const selected = document.querySelector(`input[name="q${qIdx}"]:checked`);
        if(selected && parseInt(selected.value) === item.c) {
            score++;
        }
    });
    
    // Calificación sobre base 20 (Sistema peruano habitual)
    const score20 = (score / quizData.length) * 20;
    const resultBox = document.getElementById('quiz-result');
    resultBox.style.display = 'block';
    resultBox.innerHTML = `Nota Final: ${score20.toFixed(1)} / 20 puntos (${score} de 10 correctas).<br>
    <span style="font-size:1rem; font-weight:normal;">Excelente esfuerzo colega de la I.E. Leoncio Prado. Siga innovando en sus aulas.</span>`;
}

function playAIVoice() {
    const text = document.getElementById('ai-text').textContent;
    if (!text) return;
    
    // Web Speech API - Integrada nativamente en navegadores modernos para generar Audio
    window.speechSynthesis.cancel(); // Detener audios previos
    const utterance = new SpeechSynthesisUtterance(text);
    utterance.lang = 'es-ES';
    utterance.rate = 1.0;
    
    utterance.onstart = () => { document.getElementById('audio-status').textContent = "🔊 Hablando..."; };
    utterance.onend = () => { document.getElementById('audio-status').textContent = "✓ Audio finalizado"; };
    
    window.speechSynthesis.speak(utterance);
}
